function myPromiseAll(arr) {
    let i = 0, resultArr = [];
    return new Promise((resolve, reject) => {
        arr.forEach((it, idx) => {
            Promise.resolve(it).then(res => {
                i++;
                resultArr[idx] = res;
                if (i === arr.length) return resultArr;
            }, err => reject(err));
        });
    });

  • 4.找数组中的第二大的值


// 时间复杂度控制在 O(n) - 一次遍历找出来, 不用Math.max, arr.sort
/
  @param {number[]} arr (数组长度大于0, 都为数字)
  @return {number}
 */
const findEl = (arr) => {
  // 代码
}
// 用例
const arr = [1, 5, 20, 3, 9]
const el = findEl(arr)
console.log(el);

实现如下：


const findEl = (arr) => {
  // 代码
  let largest = arr[0];
  let secondLargest = arr[0];
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] > largest) {
      secondLargest = largest;
      largest = arr[i];
    } else if (arr[i] < largest && arr[i] > secondLargest) {
      secondLargest = arr[i];
    }
  }
  return secondLargest
}


  • 5.两个版本号 version1 和 version2 ，请你比较它们的大小

// 如果 version1 > version2 返回 1，
// 如果 version1 < version2 返回 -1，
// 除此之外返回 0。
/
  @param {string} version1
  @param {string} version2
  @return {number}
 /
const compareVersion = function (version1, version2) {
   // 代码
 }
 // 用例
 // 输入：version1 = ”1.01”, version2 = ”1.001”
 // 输出：0
 // 解释：忽略前导零，”01” 和 ”001” 都表示相同的整数 ”1”
 console.log(‘res = ’, compareVersion(‘1.01’, ‘1.001’))
 // 输入：version1 = ”1.0”, version2 = ”1.0.0”
 // 输出：0
 // 解释：version1 没有指定下标为 2 的修订号，即视为 ”0”
 console.log(‘res = ’, compareVersion(‘1.0’, ‘1.0.0’))
 // 输入：version1 = ”0.1”, version2 = ”1.1”
 // 输出：-1
 // 解释：version1 中下标为 0 的修订号是 ”0”，version2 中下标为 0 的修订号是 ”1” 。0 < 1，所以 version1 < version2
 console.log(‘res = ’, compareVersion(‘0.1’, ‘1.1’))

实现如下：


const compareVersion = function (version1, version2) {
  // 代码
  const transfer = (str) => {
    let arr = str.split(‘.’);
    arr = arr.map(val => {
      return parseInt(val).toString()
    })
    return arr.join(‘.’)
  }
  let v1 = transfer(version1);
  let v2 = transfer(version2);
  if (v1 == v2) return 0
  if (v1.length !== v2,length && v1.length > v2.length) {
    v2 = v2.padEnd(‘.0’, (v1.length - v2.length) / 2)
  }
  if (v1.length !== v2,length && v2.length > v1.length) {
      v1 = v1.padEnd(‘.0’, (v2.length - v1.length) / 2)
    }
  return v1 > v2 ? 1 : -1
}


  • 6.假设你正在爬楼梯。需要 n 阶你才能到达楼顶。每次你可以爬 1 或 2 个台阶。你有多少种不同的方法可以爬到楼顶呢？

// 这题主要考察的是动态规划，斐波那契函数 以及一些相关的优化
function getStep(n) {
    let deps = [1, 2];
    for(let i = 2; i < n; i++) {
        deps[i] = deps[i-1] + deps[i-2];
    }
    return deps[n-1];
}

function transferArr(arr) {
    const result = [];
    const map = arr.reduce((prev, next) => {
        prev[next.id] = next;
        return prev
    }, {});
    for (let item of arr) {
        if (item.pid === 0) {
            result.push(item)
        }
        if (item.pid in map) {
            const parent = map[item.pid];
            parent.children = parent.children || [];
            parent.children.push(item)
        }
    }
    return result;
}


  • 8.检查字符串中左右括号是否匹配

// isBracesBalanced(‘(foo { bar (baz) [boo] })’) // true
// isBracesBalanced(‘foo { bar { baz }’)         // false
// isBracesBalanced(‘foo { (bar [baz] } )’)      // false

实现如下：


function isBracesBalanced(str) {
    const arr = [];
    const map = {
        “{“: “}”,
        “(“: “)”,
        “[“: “]”
    }
    let strArr = str.split(“”);
    let flag = true;
    strArr.forEach(it => {
        if (it === “(“ || it === “{“ || it === “[“) {
            arr.push(it);
        } else if (it === “)” || it === “}” || it === “]”) {
            if (arr.length === 0 || it !== map[arr.pop()]) {
                flag = false
            }
        }
    });
    if (arr.length > 0) {
        flag = false
    }
    return flag;
}
