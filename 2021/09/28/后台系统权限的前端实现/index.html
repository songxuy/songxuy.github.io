路由权限管理


❝

这里的方法可能不是特别好，当路由跳转的时候都会去更新当前的权限列表。判断to.name在不在权限列表中，在的话跳转，不在的话提示并且跳到第一个权限路由。


❞


  • 由于一些路由并不在菜单中，其权限就和其所在页面菜单的权限绑定

const guardRoute = (to: any, from: any, next: any) => {
  var token = localStorage.getItem(“token”)
  if (!token) {
    next(“/login”)
  } else {
    store.dispatch(“getMenuList”).then(() => {
      if (
        store.state.menuList.includes(to.name) ||
        to.name === “editorGoods” ||
        to.name === “editorMileDetail” ||
        to.name === “editorContract” ||
        to.name === “addMileDetail” ||
        to.name === “addService” ||
        to.name === “editorService” ||
        to.name === “addTemplate” ||
        to.name === “editorTemplate” ||
        to.name === “templateDetail” ||
        to.name === “copyTemplate”
      ) {
        next()
      } else {
        message.error(“暂无权限，请联系管理员”)
        next({ name: store.state.menuList[0] })
      }
    })
  }
}
