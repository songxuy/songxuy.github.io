Vue3 组件通信

三月 19, 2023
  • Vue3
❝

随着Vue3越来越成熟与普及，工作和日常学习中也需要熟悉Vue3，下面就总结下在Vue3中组件的通信方式与写法（仅以setup写法为例）。


❞

Props


/ 父组件 /
<script setup lang=“ts”>
import childDemo from ‘./views/childDemo.vue’;
import { ref } from ‘vue’;
const msg = ref(“hello world”)
</script>
<template>
  <header>
    <div class=”wrapper”>
      <childDemo :message=”msg”/>
    </div>
  </header>
</template>

/ 子组件 /
<template>
  <div></div>
</template>
<script setup lang=”ts”>
const props = defineProps({
  message: {
    type: String,
    default: ””
  }
})
</script>

$Emit


/父组件/
<script setup lang=“ts”>
import child2Demo from ‘./views/child2Demo.vue’;
const clickBack = (msg: string) => {
  console.log(msg);
}
</script>
<template>
  <header>
    <div class=”wrapper”>
      <child2Demo @myClick=”clickBack”/>
    </div>
  </header>
</template>

/子组件/
<template>
    <div>
        <button @click=”handleClick”>click</button>
    </div>
  </template>
  <script setup lang=“ts”>
  const emit = defineEmits([“myClick”]);
  const handleClick = () => {
    emit(“myClick”, “xixixi”)
  }
  </script>

expose / ref



  • 父组件获取子组件数据

/父组件/
<script setup lang=“ts”>
import child3Demo from ‘./views/child3Demo.vue’;
import { ref } from ‘vue’;
const child = ref(null)
const getChild = () => {
  console.log(child.value);
}
</script>
<template>
  <header>
    <div class=”wrapper”>
      <child3Demo ref=”child”/>
      <button @click=“getChild”>get child</button>
    </div>
  </header>
</template>

/子组件/
<script setup lang=”ts”>
  defineExpose({
    message: ’暴露给父组件的信息’,
    childClick: () => {
      console.log(“子组件方法”)
    }
  })
</script>

attrs



  • 包含父作用域里除 class 和 style 除外的非 props 属性集合

/父组件/
<script setup lang=“ts”>
import child4Demo from ‘./views/child4Demo.vue’;
</script>
<template>
  <header>
    <div class=”wrapper”>
      <child4Demo :msg=”‘123’” :titile=”‘title’” />
    </div>
  </header>
</template>

/子组件/
<script setup lang=”ts”>
import { useAttrs } from ’vue’;
const attrs = useAttrs()
console.log(attrs)
</script>

v-model



  • 一个组件可以支持多个v-model

/父组件/
<script setup lang=“ts”>
import child5Demo from ‘./views/child5Demo.vue’;
import { ref } from ‘vue’;
const key = ref(‘123’)
const value = ref(‘456’)
</script>
<template>
  <header>
    <div class=”wrapper”>
      <child5Demo v-model:key=”key” v-model:value=”value” />
      <p></p>
      <p></p>
    </div>
  </header>
</template>

/子组件/
<template>
  <button @click=”handleClick”>child click</button>
</template>
<script setup lang=”ts”>
const emit = defineEmits([‘update:key’, ’update:value’])
const handleClick = () => {
  emit(“update:key”, ”new key”);
  emit(“update:value”, ”new value”)
}
</script>

provide / inject


/父组件/
<script setup lang=“ts”>
import child6Demo from ‘./views/child6Demo.vue’;
import { provide } from ‘vue’;
provide(“message”, “父组件message”)
</script>
<template>
  <header>
    <div class=”wrapper”>
      <child6Demo />
    </div>
  </header>
</template>

/子组件/
<script setup lang=”ts”>
import { inject } from ’vue’;
console.log(inject(‘message’))
</script>

Pinia



  • 作为替代Vuex的存在，更多用法与配置参见官网

/stores/message.ts/
import { ref } from ‘vue’
import { defineStore } from ‘pinia’
export const useMessageStore = defineStore(‘message’, () => {
  const msg = ref(‘hello wolrd’)
  function msgChange(message: string) {
    msg.value = message
  }
  return { msg, msgChange }
})


/组件中使用/
<template>
  <button @click=“changeMsg”>change msg</button>
  <p></p>
</template>
<script setup lang=”ts”>
import { useMessageStore } from ”@/stores/message“;
const msgStore = useMessageStore();
const changeMsg = () => {
  msgStore.msgChange(“new message“);
}
</script>

mitt



  • Vue3中虽然没有了EventBus，但是可以使用mitt.js 原理和EventBus相同

/mitt.js/
import mitt from ‘mitt’
const mitts = mitt()
export default mitts

/子组件1/
<template>
  <button @click=“handleClick”>alert bro</button>
</template>
<script setup lang=”ts”>
import mitts from ‘@/EventBus/mitt‘;
const handleClick = () => {
  mitts.emit(‘handleChange‘)
}
</script>

/子组件2/
<script setup lang=”ts”>
import mitts from ’../EventBus/mitt‘
import { onUnmounted } from ’vue‘
const someMethed = () => {
    console.log(“get message”)
 }
mitts.on(‘handleChange‘,someMethed)
onUnmounted(()=>{
    mitts.off(‘handleChange‘,someMethed)
})
</script>
